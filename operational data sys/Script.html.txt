function calcElapsed(date,time){
  try{
    var dp=date.split('-'),tp=time.split(':');
    var entry=new Date(+dp[2],+dp[1]-1,+dp[0],+tp[0],+tp[1],+(tp[2]||0));
    var s=Math.floor(Math.max(0,Date.now()-entry)/1000);
    return Math.floor(s/3600)+':'+pad2(Math.floor(s%3600/60))+':'+pad2(s%60);
  }catch(e){return'0:00:00';}
}
function durLabel(ti,to){

}

function findActiveSubVisitor(sub){

  var ids=sub.visitIds.split(',').map(function(x){return+x.trim();}).filter(Boolean);

function initInlinePaymentModal(){
  document.getElementById('ip-amount').addEventListener('input',function(){
    var adding=parseFloat(this.value)||0;
    var currentPaid=parseFloat(document.getElementById('ip-current-paid').value)||0;
    var total=parseFloat(document.getElementById('ip-total-val').value)||0;

function renderVKPIs(){
  var tv=APP.visitors.filter(function(v){return (v.entryDate||v.date)===APP.today;});
  var active=0,mins=0,newV=0;
  tv.forEach(function(v){

  });
  var h=Math.floor(mins/60),m=mins%60;
  setText('v-kpi-hours',mins>0?(h+'س '+(m>0?m+'د':'')):'—');
  setText('v-kpi-active',active);
  setText('v-kpi-total',tv.length);
  setText('v-kpi-new',newV);
}

function renderVisitors(){
  renderVKPIs();
  var search=((document.getElementById('v-search')||{}).value||'').trim().toLowerCase();
  var today=APP.today;
  var list=APP.visitors.filter(function(v){
    if(v.subType!=='ساعات'&&v.subType!=='يومي')return false;
    var vDate=v.entryDate||v.date||'';
    if(APP.vDateFilter==='today'&&vDate!==today)return false;
    if(APP.vDateFilter==='week'){
      if(!vDate)return false;
      var d=new Date(); d.setHours(0,0,0,0);
      var ws=new Date(d); ws.setDate(d.getDate()-6);
      var p=vDate.split('-');
      var vd=new Date(+p[2],+p[1]-1,+p[0]); vd.setHours(0,0,0,0);
      if(vd<ws||vd>d)return false;
    }
    if(APP.vStatusFilter==='active'&&v.exitTime!=='')return false;
    if(APP.vStatusFilter==='done'&&v.exitTime==='')return false;
    if(search&&v.name.toLowerCase().indexOf(search)===-1&&v.phone.indexOf(search)===-1)return false;
    return true;
  });
  var tb=document.getElementById('visitors-tbody');
  if(!list.length){tb.innerHTML='<tr><td colspan="11" class="empty-state">لا توجد نتائج</td></tr>';return;}
  tb.innerHTML=list.map(function(v){
    var isA=!v.exitTime;
    var sb=isA?'<span class="badge badge-active">داخل</span>':'<span class="badge badge-done">خرج</span>';
    var pb=v.paid==='نعم'?'<span class="badge badge-paid">نعم</span>':'<span class="badge badge-unpaid">لا</span>';
    var acts='<div class="td-actions"><button class="btn btn-secondary btn-sm" onclick="openEditVisitor('+v.rowIndex+')">تعديل</button>';
    if(isA){
      acts+='<button class="btn btn-primary btn-sm" onclick="openCheckout('+v.rowIndex+')">خروج</button>';
      if(v.freezeTime){
        acts+='<button class="btn btn-secondary btn-sm" style="color:var(--warning);border-color:var(--warning);" onclick="doUnfreeze('+v.rowIndex+')">▶ استئناف</button>';
      } else {
        acts+='<button class="btn btn-secondary btn-sm" style="color:var(--text-muted);" onclick="doFreeze('+v.rowIndex+')" title="تجميد الوقت مؤقتاً">⏸</button>';
      }
    }
    acts+='</div>';
    var amtRem=parseFloat(v.amountRemaining)||0;
    var payDisplay=amtRem>0?('<span style="color:var(--warning);">₪'+amtRem+' متبقي</span>'):pb;
    var frozenBadge=v.freezeTime?'<span class="badge badge-warning" title="مجمّد عند '+v.freezeTime.slice(0,5)+'">⏸</span>':'';
    return '<tr class="'+(isA?'row-active':'')+(v.freezeTime?' row-warn':'')+'"><td>'+v.name+'</td><td>'+v.phone+'</td><td>'+(v.entryDate||v.date||'')+'</td><td>'+(v.entryTime||'').slice(0,5)+frozenBadge+'</td><td>'+(v.exitTime?v.exitTime.slice(0,5):'—')+'</td><td>'+(v.duration||'—')+'</td><td>'+v.subType+'</td><td>'+(v.cost?'₪'+v.cost:'—')+'</td><td>'+payDisplay+'</td><td>'+sb+'</td><td>'+acts+'</td></tr>';
  }).join('');
}

function renderSubTable(type,filterVal,search,tbodyId){
  var q=(search||'').trim().toLowerCase();
  var list=APP.subscriptions.filter(function(s){
    if(s.subType!==type)return false;
    if(filterVal==='active'&&s.isExpired)return false;

    if(q&&s.name.toLowerCase().indexOf(q)===-1&&s.phone.indexOf(q)===-1)return false;

function doFreeze(rowIndex){
  var v=findVisitor(rowIndex); if(!v) return;

function openCheckout(ri){
  var v=findVisitor(ri); if(!v) return;
  APP.activeVisRow=ri;APP.coDate=v.entryDate||v.date;APP.coTime=v.entryTime||v.in;
  document.getElementById('co-name').textContent=v.name;
  document.getElementById('co-subtype').textContent=v.subType;
  var rate=APP.opts.HOURLY_RATE||5;
  var dp=(APP.coDate||'').split('-'),tp=(APP.coTime||'').split(':');
  var elapsed=0;
  if(dp.length===3&&tp.length>=2){
    var entryMs=new Date(+dp[2],+dp[1]-1,+dp[0],+tp[0],+tp[1],+(tp[2]||0)).getTime();
    elapsed=Math.max(0,(Date.now()-entryMs)/3600000);
  }
  if(v.freezeTime){
    var ftp=v.freezeTime.split(':');
    var freezeMs=new Date(+dp[2],+dp[1]-1,+dp[0],+ftp[0],+ftp[1],+(ftp[2]||0)).getTime();
    elapsed=Math.max(0,(freezeMs-new Date(+dp[2],+dp[1]-1,+dp[0],+tp[0],+tp[1],+(tp[2]||0)).getTime())/3600000);
  }
  var autoCost=v.subType==='يومي'?20:parseFloat((elapsed*rate).toFixed(2));
  var autoEl=document.getElementById('co-auto-cost');if(autoEl)autoEl.textContent='₪ '+autoCost;
  var costEl=document.getElementById('co-cost');if(costEl)costEl.value=autoCost;
  var apEl=document.getElementById('co-amount-paid');if(apEl)apEl.value='';
  var remEl=document.getElementById('co-remaining');if(remEl)remEl.textContent='₪ '+autoCost;
  document.getElementById('co-paid').checked=false;
  setSel('co-paytype','');toggleAcct('co-paytype','co-account-group');
  startCoTimer();openModal('modal-checkout');
  if(apEl)setTimeout(function(){apEl.focus();},200);
}

function renderTkKPIs(){
  var today=APP.today,mins=0,count=0,cost=0;
  APP.tamkeen.forEach(function(s){
    if(s.date)return;
    var sess=APP.tkSession[s.rowIndex],liveIn=sess?sess.in:null,liveOut=sess?sess.out:null;
    var ti=liveIn||s.timeIn||'',to=liveOut||s.timeOut||'';
    var attended=!!(ti&&(s.todayDate===today||liveIn));
    if(!attended)return;
    count++;cost+=parseFloat(s.cost)||0;
    if(ti&&to){var a=ti.split(':'),b=to.split(':');var d=(+b[0]*60+ +b[1])-(+a[0]*60+ +a[1]);if(d>0)mins+=d;}
  });
  var h=Math.floor(mins/60),m=mins%60;
  setText('kpi-tk-hours',mins>0?(h+'س '+(m>0?m+'د':'')):'0س');
  setText('kpi-tk-count',count);setText('kpi-tk-cost','₪'+cost.toFixed(2));
}
function doFreezeTk(baseRow,attRow){
  if(!attRow){showToast('يرجى الانتظار لحظة...','warn');return;}
  google.script.run
    .withSuccessHandler(function(res){
      if(!res.success){showToast(res.error,'error');return;}
      var s=APP.tamkeen.find(function(x){return(x.rowIndex||x.ri)===baseRow;});if(s)s.freezeTime=res.freezeTime;
      var sess=APP.tkSession[baseRow];if(sess)sess.freeze=res.freezeTime;
      renderTamkeen();showToast('⏸ تم تجميد وقت الطالب — '+res.durationSoFar,'success');
    })
    .withFailureHandler(function(e){showToast('خطأ: '+e.message,'error');})
    .processRequest('freezeTamkeenTime',{attendanceRowIndex:attRow});
}
function doUnfreezeTk(baseRow,attRow){
  if(!attRow){showToast('يرجى الانتظار لحظة...','warn');return;}
  google.script.run
    .withSuccessHandler(function(res){
      if(!res.success){showToast(res.error,'error');return;}
      var s=APP.tamkeen.find(function(x){return(x.rowIndex||x.ri)===baseRow;});if(s){s.freezeTime='';s.timeIn=res.newEntryTime;}
      var sess=APP.tkSession[baseRow];if(sess){sess.freeze='';sess.in=res.newEntryTime;}
      renderTamkeen();showToast('▶ تم استئناف الوقت للطالب','success');
    })

}
function tkCheckin(ri){
  APP.tkSession[ri]={in:nowTime(),out:null,attendanceRowIndex:null};renderTamkeen();
  google.script.run
    .withSuccessHandler(function(res){
      if(!res.success){showToast('فشل: '+res.error,'error');delete APP.tkSession[ri];renderTamkeen();return;}
      APP.tkSession[ri].attendanceRowIndex=res.newRowIndex;APP.tkSession[ri].freeze='';
      var s=APP.tamkeen.find(function(x){return(x.rowIndex||x.ri)===ri;});
      if(s){s.attendanceRowIndex=res.newRowIndex;s.todayDate=APP.today;s.timeIn=APP.tkSession[ri].in;s.freezeTime='';}
      renderTamkeen();
    })
    .withFailureHandler(function(){showToast('فشل الدخول','error');delete APP.tkSession[ri];renderTamkeen();})
    .processRequest('tamkeenCheckin',{rowIndex:ri});
  var s=APP.tamkeen.find(function(x){return(x.rowIndex||x.ri)===ri;});
  if(s)showToast('تم تسجيل دخول '+s.name+' ✓','success');
  setTimeout(function(){loadDashboard(APP.dashPeriod);},2000);
}
function tkCheckout(ri){
  var sess=APP.tkSession[ri];
  var student=APP.tamkeen.find(function(x){return(x.rowIndex||x.ri)===ri;});
  var attIdx=(sess&&sess.attendanceRowIndex)||(student&&student.attendanceRowIndex)||null;
  if(!attIdx){showToast('جاري الحفظ — حاول مرة أخرى...','warn');setTimeout(function(){tkCheckout(ri);},1000);return;}
  var nowT=nowTime();
  if(sess)sess.out=nowT;if(student)student.timeOut=nowT;renderTamkeen();
  google.script.run
    .withSuccessHandler(function(res){
      if(!res.success){showToast('فشل الخروج: '+res.error,'error');if(sess)sess.out=null;if(student)student.timeOut='';renderTamkeen();return;}
      if(student){student.timeOut=res.timeOut;student.duration=res.duration;student.durationH=res.durationH;student.cost=String(res.cost||0);student.todayDate=APP.today;}
      renderTamkeen();renderTkKPIs();
      showToast('تم تسجيل خروج '+(student?student.name:'')+' ✓','success');
      setTimeout(function(){loadDashboard(APP.dashPeriod);},1500);
    })
    .withFailureHandler(function(e){showToast('فشل الخروج: '+(e.message||''),'error');if(sess)sess.out=null;if(student)student.timeOut='';renderTamkeen();})
    .processRequest('tamkeenCheckout',{attendanceRowIndex:attIdx});
}

}
function initEditTamkeenModal(){
  document.getElementById('btn-et-submit').onclick=function(){

    btn.disabled=true;btn.textContent='جاري...';
    var attIdx=s&&s.attendanceRowIndex?s.attendanceRowIndex:(APP.tkSession[ri]&&APP.tkSession[ri].attendanceRowIndex);
    var needsTimeUpdate=attIdx&&(ti||to),pending=needsTimeUpdate?2:1;
    function done(){pending--;if(pending<=0){closeModal('modal-edit-tamkeen');renderTamkeen();showToast('تم الحفظ ✓','success');}}
    google.script.run
      .withSuccessHandler(function(res){btn.disabled=false;btn.textContent='حفظ التعديلات';if(!res.success){showToast(res.error,'error');return;}if(s)Object.assign(s,{group:data.group,name:data.name,phone:data.phone,email:data.email,notes:data.notes});done();})

    if(needsTimeUpdate){
      google.script.run
        .withSuccessHandler(function(res){
          if(!res.success){showToast('تحذير: فشل تحديث الأوقات','error');done();return;}

        .processRequest('updateTamkeenAttendance',{attendanceRowIndex:attIdx,timeIn:ti?ti+':00':'',timeOut:to?to+':00':''});
    }
  };
}

    if(remEl){remEl.textContent='₪ '+r.toFixed(2);remEl.style.color=r>0?'var(--danger)':'var(--success)';}
  }
  if(costEl)costEl.oninput=_updateRem;if(paidEl)paidEl.oninput=_updateRem;
  if(paytEl)paytEl.onchange=function(){if(acctGrp)acctGrp.style.display=paytEl.value==='تحويل بنكي'?'':'none';};
  var submitBtn=document.getElementById('btn-fe-submit');
  if(!submitBtn) return;
  submitBtn.addEventListener('click',function(){
    var btn=this;
    var ri=+document.getElementById('fe-row').value;
    var name=document.getElementById('fe-name').value.trim();
    var phone=document.getElementById('fe-phone').value.trim();
    var sub=document.getElementById('fe-subtype')?document.getElementById('fe-subtype').value:'';
    var dateYMD=document.getElementById('fe-date').value;
    var dateDMY=dateYMD?dateYMD.split('-').reverse().join('-'):'';
    var cost=parseFloat(document.getElementById('fe-cost').value)||0;
    var paid=parseFloat(document.getElementById('fe-paid').value)||0;
    var rem=Math.max(0,cost-paid);
    var payType=(document.getElementById('fe-paytype')||{}).value||'';
    var account=(document.getElementById('fe-account')||{}).value||'';
    if(!name||!dateDMY){showToast('يرجى تعبئة الاسم والتاريخ','error');return;}
    btn.disabled=true;btn.textContent='جاري...';
    google.script.run
      .withSuccessHandler(function(res){
        btn.disabled=false;btn.textContent='حفظ التعديلات';
        if(!res.success){showToast(res.error||'فشل','error');return;}
        closeModal('modal-future-edit');
        showToast('تم حفظ التعديلات ✓','success');
        loadFutureBookings();
        if(cost>0)setTimeout(function(){loadDashboard(APP.dashPeriod);},800);
      })
      .withFailureHandler(function(e){btn.disabled=false;btn.textContent='حفظ التعديلات';showToast(e.message,'error');})
      // يستخدم updateFutureBooking وليس updateVisitor
      .processRequest('updateFutureBooking',{
        rowIndex:ri, name:name, phone:phone, subType:sub,
        date:dateDMY, cost:cost, amountPaid:paid, amountRemaining:rem,
        payType:payType, account:account
      });
  });
}

// ── Role-based UI ─────────────────────────────────────────────
function _applyRoleUI(){
  var role=sessionStorage.getItem('aw_role')||'admin';
  if(role==='social'){
    document.querySelectorAll('.nav-tab').forEach(function(t){if(t.dataset.page!=='rooms')t.style.display='none';});
    var nav=document.querySelector('.navbar-inner');
    if(nav&&!document.getElementById('nav-logout')){
      var lb=document.createElement('button');lb.id='nav-logout';lb.textContent='خروج ↩';
      lb.style.cssText='margin-right:auto;padding:.28rem .65rem;background:transparent;border:1.5px solid rgba(255,255,255,.4);border-radius:6px;color:#fff;font-family:Tajawal,sans-serif;font-size:.75rem;cursor:pointer;';
      lb.onclick=function(){sessionStorage.clear();var ov=document.getElementById('login-overlay');if(ov){ov.style.display='flex';document.getElementById('l-name').value='';document.getElementById('l-err').style.display='none';}};
      nav.appendChild(lb);
    }
    showPage('rooms');
  }
}
